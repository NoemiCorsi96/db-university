#2. selezionare tutti gli studenti nati nel 1990 
SELECT *
FROM `students`
WHERE YEAR(`date_of_birth`) = 1990;

#3.Selezionare tutti i corsi che valgono più di 10 crediti
SELECT *
FROM `courses`
WHERE `cfu` > 10;

#4. Selezionare tutti gli studenti che hanno più di 30 anni
SELECT *
FROM `students`
WHERE TIMESTAMPDIFF(YEAR, `date_of_birth`, CURDATE()) > 30

##5.Selezionare tutti i corsi del primo semestre del primo anno di un qualsiasi corso di laurea (286)
SELECT * 
FROM `courses`
WHERE `period`= 'I semestre' and `year`=1;
##6. Selezionare tutti gli appelli d'esame che avvengono nel pomeriggio (dopo le 14) del 20/06/2020 (21)
SELECT *
FROM `exams`
WHERE DATE(`date`) = '2020-06-20'
  AND TIME(`hour`) > '14:00:00';

##7. Selezionare tutti i corsi di laurea magistrale (38)
SELECT *
FROM `degrees`
WHERE `level` = 'magistrale';
##8. Da quanti dipartimenti è composta l'università? (12)

FROM `departments`;
##9. Quanti sono gli insegnanti che non hanno un numero di telefono?
SELECT COUNT(*) AS `insegnati_notelefono`
FROM `teachers`
WHERE `phone`= '' OR `phone` IS NULL;
